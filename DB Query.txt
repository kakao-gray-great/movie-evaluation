// USER 테이블 만들기
CREATE TABLE USER (
	userID varchar(20) PRIMARY KEY,
	userPassword varchar(64),
	userEmail varchar(50),
	userEmailHash varchar(64),
	userEmailChecked boolean
);

// 평가 테이블 만들기
CREATE TABLE EVALUATION (
	evaluationID int PRIMARY KEY AUTO_INCREMENT,
	userID varchar(20),
	movieTitle varchar(50),
	directorName varchar(20),
	genre varchar(10),
	evaluationTitle varchar(50),
	evaluationContent varchar(2048),
	totalScore int,
	storyScore int,
	videoScore int,
	characterScore int,
	likeCount int
);

// 좋아요 테이블 만들기
CREATE TABLE LIKEY (
	userID varchar(20),
	evaluationID int,
	userIP varchar(50)
);

// 영화 목록 테이블 만들기
CREATE TABLE MOVIE (
	movieID int PRIMARY KEY AUTO_INCREMENT,
	movieTitle varchar(50),
	directorName varchar(20),
	genre varchar(10),
	fileName varchar(30)
);

// 좋아요 테이블 primary key 설정
ALTER TABLE LIKEY ADD PRIMARY KEY (userID, evaluationID);
